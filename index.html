// api/pacientes.js - Lista pacientes y fisios para el panel const AIRTABLE_TOKEN = process.env.AIRTABLE_TOKEN; const FISIO_PASSWORD = process.env.FISIO_PASSWORD || 'fisio2024'; const BASE_ID = 'appbK09V4X3pPIai3'; const PACIENTES_TABLE = 'tbldBVgClS4HY2mOJ'; const FISIO_TABLE = 'tbl2mLUrnaKCFTs6g'; const BIBLIOTECA_TABLE = 'tbloqn3ts872ueJSE'; export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'GET, OPTIONS'); if (req.method === 'OPTIONS') return res.status(200).end(); const { pwd } = req.query; if (pwd !== FISIO_PASSWORD) return res.status(401).json({ ok: false, error: 'Contraseña incorrecta' }); try { const [pacRes, fisioRes, bibRes] = await Promise.all([ fetch(`https://api.airtable.com/v0/${BASE_ID}/${PACIENTES_TABLE}?fields[]=FULL NAME&fields[]=EMAIL&fields[]=WHATSAPP&sort[0][field]=FULL NAME&sort[0][direction]=asc&pageSize=100`, { headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}` } }), fetch(`https://api.airtable.com/v0/${BASE_ID}/${FISIO_TABLE}?fields[]=Name&fields[]=NºColegiado&fields[]=Foto`, { headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}` } }), fetch(`https://api.airtable.com/v0/${BASE_ID}/${BIBLIOTECA_TABLE}?fields[]=Nombre&fields[]=Descripcion&fields[]=Zona&fields[]=YouTubeURL&fields[]=Imagenes&sort[0][field]=Nombre&sort[0][direction]=asc&pageSize=100`, { headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}` } }) ]); const [pacData, fisioData, bibData] = await Promise.all([pacRes.json(), fisioRes.json(), bibRes.json()]); return res.status(200).json({ ok: true, pacientes: (pacData.records || []).map(r => ({ id: r.id, nombre: r.fields['FULL NAME'] || r.fields['EMAIL'] || '—', email: r.fields['EMAIL'] || '', telefono: r.fields['WHATSAPP'] || '' })), fisios: (fisioData.records || []).map(r => ({ id: r.id, nombre: r.fields['Name'] || '—', colegiado: r.fields['NºColegiado'] || '', foto: r.fields['Foto']?.[0]?.url || '' })), biblioteca: (bibData.records || []).map(r => ({ id: r.id, nombre: r.fields['Nombre'] || '—', descripcion: r.fields['Descripcion'] || '', zona: r.fields['Zona'] || '', youtubeUrl: r.fields['YouTubeURL'] || '', imagen: r.fields['Imagenes']?.[0]?.url || '' })) }); } catch (err) { return res.status(500).json({ ok: false, error: 'Error interno' }); } }
